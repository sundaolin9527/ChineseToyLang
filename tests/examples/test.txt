    若 (起始下标 >= 结束下标){
        返;
    }

    基准值 = 待排序数组[结束下标];
    分割点下标 = 起始下标;
    
    遍 (令 i = 起始下标; i< (结束下标-1); i+=1){
        若 (待排序数组[i] <= 基准值){
            交换(待排序数组[i],待排序数组[分割点下标]);
            分割点下标 += 1;
        }
    }
    
    交换(待排序数组[分割点下标],待排序数组[结束下标]);
    
    快速排序(待排序数组, 起始下标, 分割点下标 - 1);
    快速排序(待排序数组, 分割点下标 + 1, 结束下标);
}
